116 bool ConfigServer::OnFrame(std::shared_ptr<Frame> a_Frame) {
117  // Dispatch
118  assert(a_Frame);
119  auto l_ConfigFrame = std::dynamic_pointer_cast<ConfigFrame>(a_Frame);
120  assert(l_ConfigFrame);
121  switch (l_ConfigFrame->GetConfigFrameType()) {
122  case CONFIG_FRAME_GATEWAY_CLIENT_CLEANUP: {
123  m_GatewayClientHandlerCollection->CleanAll();
124  break;
125  }
126  case CONFIG_FRAME_GATEWAY_CLIENT_CREATE: {
127  auto l_GatewayClientCreate = std::dynamic_pointer_cast<GatewayClientCreate>(l_ConfigFrame);
128  m_GatewayClientHandlerCollection->CreateClient(l_GatewayClientCreate->GetReferenceNbr(),
129  l_GatewayClientCreate->GetRemoteAddress(),
130  l_GatewayClientCreate->GetRemotePortNbr());
131  break;
132  }
133  case CONFIG_FRAME_GATEWAY_CLIENT_DESTROY: {
134  auto l_GatewayClientDestroy = std::dynamic_pointer_cast<GatewayClientDestroy>(l_ConfigFrame);
135  m_GatewayClientHandlerCollection->DestroyClient(l_GatewayClientDestroy->GetReferenceNbr());
136  break;
137  }
138  case CONFIG_FRAME_HDLCD_CLIENT_CLEANUP: {
139  m_HdlcdClientHandlerCollection->CleanAll();
140  break;
141  }
142  case CONFIG_FRAME_HDLCD_CLIENT_CREATE: {
143  auto l_HdlcdClientCreate = std::dynamic_pointer_cast<HdlcdClientCreate>(l_ConfigFrame);
144  m_HdlcdClientHandlerCollection->CreateHdlcdClient(l_HdlcdClientCreate->GetRemoteAddress(), l_HdlcdClientCreate->GetTcpPortNbr(), l_HdlcdClientCreate->GetSerialPortNbr());
145  break;
146  }
147  case CONFIG_FRAME_HDLCD_CLIENT_DESTROY: {
148  auto l_HdlcdClientCreate = std::dynamic_pointer_cast<HdlcdClientCreate>(l_ConfigFrame);
149  m_HdlcdClientHandlerCollection->DestroyHdlcdClient(l_HdlcdClientCreate->GetSerialPortNbr());
150  break;
151  }
152  case CONFIG_FRAME_HDLCD_CLIENT_SUSPEND: {
153  auto l_HdlcdClientSuspend = std::dynamic_pointer_cast<HdlcdClientSuspend>(l_ConfigFrame);
154  m_HdlcdClientHandlerCollection->SuspendHdlcdClient(l_HdlcdClientSuspend->GetSerialPortNbr());
155  break;
156  }
157  case CONFIG_FRAME_HDLCD_CLIENT_RESUME: {
158  auto l_HdlcdClientResume = std::dynamic_pointer_cast<HdlcdClientResume>(l_ConfigFrame);
159  m_HdlcdClientHandlerCollection->ResumeHdlcdClient(l_HdlcdClientResume->GetSerialPortNbr());
160  break;
161  }
162  default:
163  // Ignore
164  break;
165  } // switch
166 
167  return true;
168 }
